def isSameAfterReversals(num):
    reverse = doThis(num)
    reverse2 = doThis(reverse)

    return True if reverse2 == num else False

def doThis(num):
    ans = 0
    while (num > 0):
        temp = num % 10
        num = num // 10
        ans = ans*10 + temp
    return ans
